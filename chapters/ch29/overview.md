```python
import pandas as pd
```

### νƒκµ¬ 2: μ μλ“¤μ„ μ„ν• κΈ°λ„ λ°μ΄ν„° ν›λ ¨/ν…μ¤νΈ λ¶„ν•  (train_test_split, seed)

```python
import pandas as pd
import numpy as np
from chapters.ch29.ordination_data import OrdinationDataGenerator # λ™μΌ λ°μ΄ν„° μƒμ„±κΈ° μ‚¬μ©
from chapters.ch29.train_test_splitter import TrainTestSplitter
from chapters.ch29.reproducible_split import ReproducibleSplitter

# μ μ λ°μ΄ν„° μƒμ„± (κ°€μ •)
data_gen = OrdinationDataGenerator()
df_disciples = data_gen.generate_ordination_data()

print("π›΅οΈ μ μ λ°μ΄ν„° μ”μ•½:\n", df_disciples.head())

# ν›λ ¨/ν…μ¤νΈ λ°μ΄ν„° λ¶„ν• 
splitter = TrainTestSplitter(df_disciples)
train_df, test_df = splitter.split_data(test_size=0.2, random_state=29)

print("\n--- ν›λ ¨/ν…μ¤νΈ λ°μ΄ν„° λ¶„ν•  ν›„ ---")
print(f"ν›λ ¨ μ„ΈνΈ ν¬κΈ°: {len(train_df)}, ν…μ¤νΈ μ„ΈνΈ ν¬κΈ°: {len(test_df)}")
print("ν›λ ¨ μ„ΈνΈ (μΌλ¶€):\n", train_df.head())

# μ¬ν„ κ°€λ¥ν• λ°μ΄ν„° λ¶„ν•  μ‹μ—°
repro_splitter = ReproducibleSplitter(df_disciples)
train_df_repro1, test_df_repro1 = repro_splitter.split_data_reproducibly(test_size=0.2, random_state=123)
train_df_repro2, test_df_repro2 = repro_splitter.split_data_reproducibly(test_size=0.2, random_state=123)

print("\n--- μ¬ν„ κ°€λ¥ν• λ¶„ν•  ν™•μΈ (λ‘ ν›λ ¨ μ„ΈνΈ λ™μΌ μ—¬λ¶€) ---")
print(f"λ‘ ν›λ ¨ μ„ΈνΈκ°€ λ™μΌν•κ°€μ”? {train_df_repro1.equals(train_df_repro2)}")
```
